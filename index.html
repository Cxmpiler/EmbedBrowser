function setFrame(targetUrl) { currentUrlDisplay.textContent = targetUrl; status.textContent = "Loading..."; frameWrap.innerHTML = ""; const iframe = document.createElement("iframe"); // We MUST encode the URL so the Worker receives the full string correctly const proxyUrl = "https://embedstuff.cxmplate.workers.dev/?url=" + encodeURIComponent(targetUrl); iframe.src = proxyUrl; iframe.style.width = "100%"; iframe.style.height = "100%"; iframe.onload = () => status.textContent = "Loaded"; frameWrap.appendChild(iframe); }
